import { Component, Input } from "@angular/core";

import { Task } from "../../../../shared/models/task.model";

@Component({
	selector: 'task-list-grid-view',
	templateUrl: './task-list-grid-view.component.html',
	styleUrls: ['./task-list-grid-view.component.scss']
})
export class TaskListGridViewComponent {
	@Input('tasksDueToday') tasksDueToday: Task[];
	@Input('tasksNotDueToday') tasksNotDueToday: Task[];
	@Input('tasksDelayed') tasksDelayed: Task[];
	@Input('tasksFinished') tasksFinished: Task[];
}

<div *ngIf="tasksDueToday.length">
	<h2>Encerram hoje</h2>
	<task-grid [rowData]="tasksDueToday"></task-grid>
</div>

<div *ngIf="tasksNotDueToday.length">
	<h2>Proximas a encerrar</h2>
	<task-grid [rowData]="tasksNotDueToday"></task-grid>
</div>

<div *ngIf="tasksDelayed.length">
	<h2>Atrasadas</h2>
	<task-grid [rowData]="tasksDelayed"></task-grid>
</div>

<div *ngIf="tasksFinished.length">
	<h2>Últimas Concluídas</h2>
	<task-grid [rowData]="tasksFinished"></task-grid>
</div>
